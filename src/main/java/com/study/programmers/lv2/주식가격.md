import java.util.*;

class Solution {
    
    public int[] solution(int[] prices) {
        Stack<Integer> st = new Stack<>();
        int[] answer = new int[prices.length];
        
        for (int i = 0; i < prices.length; i++) {
            while (!st.isEmpty() && prices[st.peek()] > prices[i]) {
                int top = st.pop();
                answer[top] = i - top;
            }
            st.push(i);
        }
        
        while (!st.isEmpty()) {
            int top = st.pop();
            answer[top] = prices.length - 1 - top;
        }
        return answer;
    }
}
